### 📋 HWPX 표 데이터 Excel 추출 (Python)
```python
import zipfile
import xml.etree.ElementTree as ET
import openpyxl
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side
from openpyxl.utils import get_column_letter

hwpx_path = r"c:\Users\user\Desktop\ai사무자동화\예제파일\11_한글_문서_실습\01_hwpx_추출\월간회의록.hwpx"

with zipfile.ZipFile(hwpx_path, 'r') as z:
    xml_data = z.read("Contents/section0.xml")
    root = ET.fromstring(xml_data)

    'hp': 'http://www.hancom.co.kr/hwpml/2011/paragraph',
    'hs': 'http://www.hancom.co.kr/hwpml/2011/section'

rows_data = []

rows = tbl.findall('.//hp:tr', ns)
for row in rows:
    row_cells = []
    cells = row.findall('.//hp:tc', ns)
    for cell in cells:

        for p in cell.findall('.//hp:p', ns):
            p_text = []
            for t in p.findall('.//hp:t', ns):

                    p_text.append(t.text)
            if p_text:
                cell_texts.append("".join(p_text))

    rows_data.append(row_cells)

ws.views.sheetView[0].showGridLines = True

            else:
                cell.alignment = align_left

```

### 📋 HWP OLE 이진 데이터 수정 및 HWPX 생성 (Python)
```python
import zlib
import struct
import pythoncom

def modify_section0(data):
    # Decompress using raw deflate (wbits=-15)
    decompressed = zlib.decompress(data, -15)
    
    # Parse records
    offset = 0
    records = []
    while offset < len(decompressed):
        if offset + 4 > len(decompressed):
            break
        header = struct.unpack("<I", decompressed[offset:offset+4])[0]
        tag_id = header & 0x3FF
        level = (header >> 10) & 0x3FF
        size = (header >> 20) & 0xFFF
        offset += 4
        if size == 0xFFF:
            size = struct.unpack("<I", decompressed[offset:offset+4])[0]
            offset += 4
        record_data = decompressed[offset:offset+size]
        offset += size
        records.append({"tag_id": tag_id, "level": level, "data": record_data})
        
    # Replacement mapping for 사내안내문
    replacements = {
        51: "2025\r",
        61: "04\r",
        71: "02\r",
        137: "인사팀\r",
        178: "사내 교육 프로그램 안내\r",
        182: "  전 직원 역량 강화를 위한 사내 교육 프로그램을 아래와 같이 \r",
        186: "  안내드리오니 일정을 확인하시어 신청해 주시기 바랍니다.\r",
        200: "1. 교육명 : AI 사무 자동화 실무 과정\r",
        204: " - 일시 및 장소: 2025년 4월 7일(월) ~ 4월 11일(금), 본사 3층 교육장\r",
        217: "2. 교육대상 : 전 직원 (부서별 선착순 신청)\r",
        221: "3. 신청기한 : 2025년 4월 2일(수) 18:00까지\r",
        225: "4. 신청방법 : 사내 포털 → 교육신청 메뉴\r"
    }
    
    for idx, new_text in replacements.items():
        new_data = new_text.encode('utf-16le')
        records[idx]["data"] = new_data
        
        # Update character count in preceding PARA_HEADER
        prev_idx = idx - 1
        while prev_idx >= 0 and records[prev_idx]["tag_id"] != 66:
            prev_idx -= 1
            
        if prev_idx >= 0:
            header_rec = records[prev_idx]
            original_field = struct.unpack("<I", header_rec["data"][:4])[0]
            new_char_len = len(new_text)
            if original_field >= 2147483648:
                new_field = 0x80000000 | new_char_len
            else:
                new_field = new_char_len
            header_rec["data"] = struct.pack("<I", new_field) + header_rec["data"][4:]
            
    # Reassemble decompressed bytes
    new_decompressed = bytearray()
    for r in records:
        tag_id = r["tag_id"]
        level = r["level"]
        r_data = r["data"]
        size = len(r_data)
        
        if size >= 0xFFF:
            header = (tag_id & 0x3FF) | ((level & 0x3FF) << 10) | (0xFFF << 20)
            new_decompressed.extend(struct.pack("<II", header, size))
        else:
            header = (tag_id & 0x3FF) | ((level & 0x3FF) << 10) | ((size & 0xFFF) << 20)
            new_decompressed.extend(struct.pack("<I", header))
            
        new_decompressed.extend(r_data)
        
    # Compress using raw deflate
    compressor = zlib.compressobj(wbits=-15)
    new_compressed = compressor.compress(bytes(new_decompressed)) + compressor.flush()
    return new_compressed

def copy_storage(src, dest):
    for stat in src.EnumElements():
        name = stat[0]
        type = stat[1]
        if type == 1: # STGTY_STORAGE
            sub_src = src.OpenStorage(name, None, 0x00000000 | 0x00000010)
            sub_dest = dest.CreateStorage(name, 0x1012, 0, 0)
            copy_storage(sub_src, sub_dest)
        elif type == 2: # STGTY_STREAM
            sub_src = src.OpenStream(name, None, 0x00000000 | 0x00000010)
            sub_dest = dest.CreateStream(name, 0x1012, 0, 0)
            
            data = sub_src.Read(stat[2])
            if name == "Section0":
                data = modify_section0(data)
            elif name == "PrvText":
                data = modify_prv_text(data)
                
            sub_dest.Write(data)
```
